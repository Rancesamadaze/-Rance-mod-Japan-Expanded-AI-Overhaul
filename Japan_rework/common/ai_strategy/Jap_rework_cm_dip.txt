JAP_cm_reject_non_prc_ally_war_calls = {
	allowed = {
		always = yes
	}
	reversed = yes
	enable = {
		country_exists = JAP
		has_war = yes
		is_in_faction_with = JAP
		NOT = { tag = JAP }
		NOT = { tag = PRC }
		JAP = {
			has_government = communism
		}
	}
	enable_reverse = {
		JAP = {
			original_tag = JAP
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = diplo_action_acceptance
		id = JAP
		target = call_allies
		value = -99999
	}
	ai_strategy = {
		type = diplo_action_desire
		id = JAP
		target = join_allies
		value = -99999
	}
}
