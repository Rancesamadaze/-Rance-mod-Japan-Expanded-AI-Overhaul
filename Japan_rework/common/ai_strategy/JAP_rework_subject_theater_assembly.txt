	allowed = {
		always = yes
	}

	enable = {
		JAP_rework_is_japanese_special_subject = yes
		JAP = {
			has_country_flag = JAP_rework_subject_assembly_southeast_asia_theater_active
			JAP_rework_subject_assembly_southeast_asia_theater_staging_available = yes
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = put_unit_buffers
		order_id = 62302
		ratio = 0.30
		states = {
			325 # 云南
			599 # 广西
			594 # 南宁
			592 # 广州
			593 # 广东
			591 # 海南
		}
		area = JAP_rework_southeast_asia_theater
		subtract_invasions_from_need = yes
		subtract_fronts_from_need = yes
	}
}
