JAP_ai_general_apply_current_leader_by_owner_difficulty = {
	if = {
		limit = {
			is_army_leader = yes
			OWNER = {
				tag = JAP
				Rance_is_jap_ai = yes
			}
		}
		if = {
			limit = {
				OWNER = { Rance_nd_more_than_lunatic = yes }
			}
			JAP_ai_general_apply_current_leader_exph_tier = yes
		}
		else_if = {
			limit = {
				OWNER = { Rance_nd_more_than_normal = yes }
			}
			JAP_ai_general_apply_current_leader_hl_tier = yes
		}
		else = {
			JAP_ai_general_apply_current_leader_en_tier = yes
		}
	}
}

JAP_ai_dynamic_grant_refresh_multipliers = {
	set_variable = { JAP_ai_dynamic_base_multiplier = 1 }
	set_variable = { JAP_ai_boss_bonus_multiplier = 0 }

	if = {
		limit = { has_war = yes }
		set_variable = { JAP_ai_dynamic_base_multiplier = 2 }

		if = {
			limit = { surrender_progress > 0.05 }
			set_variable = { JAP_ai_dynamic_base_multiplier = 3 }
		}

		if = {
			limit = {
				has_game_rule = {
					rule = rance_japan_ai_boss_mode
					option = ENABLED
				}
			}

			every_enemy_country = {
				limit = {
					is_ai = no
					is_major = yes
				}
				JAP = { add_to_variable = { JAP_ai_boss_bonus_multiplier = 2 } }
			}
			every_enemy_country = {
				limit = {
					is_ai = no
					is_major = no
				}
				JAP = { add_to_variable = { JAP_ai_boss_bonus_multiplier = 1 } }
			}
			every_enemy_country = {
				limit = {
					is_ai = yes
					is_major = yes
				}
				JAP = { add_to_variable = { JAP_ai_boss_bonus_multiplier = 1 } }
			}
		}
	}

	set_variable = {
		var = JAP_ai_dynamic_final_multiplier
		value = JAP_ai_dynamic_base_multiplier
	}
	add_to_variable = {
		var = JAP_ai_dynamic_final_multiplier
		value = JAP_ai_boss_bonus_multiplier
	}
}
