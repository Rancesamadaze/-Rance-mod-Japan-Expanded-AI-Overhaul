JAP_ai_navy_create_one_formal_carrier = {
    if = {
        limit = {
            OR = {
                JAP = { has_country_flag = JAP_has_cr_ship_design_cv_akagi }
                JAP = { has_country_flag = JAP_has_cr_ship_design_cv_shokaku }
                JAP = { has_country_flag = JAP_has_cr_ship_design_cv_shinano }
                JAP = { has_country_flag = JAP_has_cr_ship_design_cv_hakuryu }
            }
        }
        meta_effect = {
            text = {
                create_ship = {
                    type = ship_hull_carrier_[JAP_ai_navy_text_formal_carrier_hull]
                    equipment_variant = "[JAP_ai_navy_text_formal_carrier_variant]"
                    creator = JAP
                }
            }
            JAP_ai_navy_text_formal_carrier_hull = "[GetJAP_ai_navy_carrier_hull]"
            JAP_ai_navy_text_formal_carrier_variant = "[GetJAP_ai_navy_carrier_variant]"
        }
    }
}
