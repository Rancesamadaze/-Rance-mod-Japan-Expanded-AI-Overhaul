JAP_rework_add_co_prosperity_governance = {
	if = {
		limit = {
			JAP_rework_is_japanese_special_subject = yes
			NOT = { has_dynamic_modifier = { modifier = JAP_rework_co_prosperity_governance_dm } }
		}
		custom_effect_tooltip = JAP_rework_add_co_prosperity_governance_modifier_tt
		hidden_effect = {
			add_dynamic_modifier = { modifier = JAP_rework_co_prosperity_governance_dm }
		}
	}
}

JAP_rework_subject_apply_co_prosperity_governance_effect = {
	if = {
		limit = {
			JAP_rework_is_japanese_special_subject = yes
		}
		JAP_rework_add_co_prosperity_governance = yes
		set_country_flag = JAP_rework_received_co_prosperity_governance_reward
	}
}
