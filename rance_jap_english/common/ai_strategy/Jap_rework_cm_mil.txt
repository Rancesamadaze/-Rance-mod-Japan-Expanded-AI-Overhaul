## 后中国/战后外线门槛统一使用：has_completed_focus = JAP_the_pacific_union_of_soviet_socialist_republics
## 不包含北美远征、北方对苏和第二共产路线内容。

## 朝鲜登陆后 rush：登陆后 30 天内向登陆州与邻近州补兵强推。
JAP_lecture_korea_landing_followup = {
	allowed = {
		original_tag = JAP
	}
	enable = {
		Rance_is_jap_lecture_ai = yes
		has_completed_focus = JAP_reclaim_lost_territories
		any_enemy_country = { original_tag = KOR }
		has_country_flag = {
			flag = JAP_lecture_korea_landing_boost
			days < 30
		}
		JAP_lecture_home_islands_crisis = no
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = front_unit_request
		country_trigger = {
			original_tag = KOR
		}
		state_trigger = {
			OR = {
				has_state_flag = JAP_lecture_korea_landing_priority
				any_neighbor_state = {
					has_state_flag = JAP_lecture_korea_landing_priority
				}
			}
		}
		value = 20
	}

	ai_strategy = {
		type = front_control
		country_trigger = {
			original_tag = KOR
		}
		state_trigger = {
			OR = {
				has_state_flag = JAP_lecture_korea_landing_priority
				any_neighbor_state = {
					has_state_flag = JAP_lecture_korea_landing_priority
				}
			}
		}
		ratio = 0
		priority = 700
		ordertype = front
		execution_type = rush_weak
		execute_order = yes
		manual_attack = yes
	}
}
