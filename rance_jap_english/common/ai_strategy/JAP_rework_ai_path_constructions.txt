Rance_JAP_rework_constructions_industrial_complex_dynamic = {
	allowed = {
		original_tag = JAP
	}
	enable = {
		check_variable = { num_of_civilian_factories > 199 }
		check_variable = { num_of_military_factories > 149 }
		check_variable = { num_of_naval_factories > 99 }
		JAP_ai_buildings_has_industrial_complex_slot = yes
		JAP_ai_buildings_industrial_complex_low = yes
	}
	abort = {
		OR = {
			check_variable = { num_of_civilian_factories < 200 }
			check_variable = { num_of_military_factories < 150 }
			check_variable = { num_of_naval_factories < 100 }
			JAP_ai_buildings_has_industrial_complex_slot = no
			JAP_ai_buildings_industrial_complex_high = yes
		}
	}

	ai_strategy = {
		type = building_target
		id = industrial_complex
		value = 9999
	}

	ai_strategy = {
		type = building_target
		id = arms_factory
		value = -9999
	}

	ai_strategy = {
		type = building_target
		id = dockyard
		value = -9999
	}
}
