## 皇道派外交 AI
##
## 路线判定规则：
## - 本文件用于皇道派专属外交策略。
## - `allowed` 只做粗筛，例如国家 tag、original_tag、是否需要反向策略等。
## - 皇道派路线判定统一写在 `enable` 中，调用现成 scripted trigger：
##
##   Rance_is_jap_kodoha_ai = yes
##
## - 不要在本文件反复展开 `JAP_AI_RANDOM_NEUTRALITY` 或 `JAP_ai_behavior = NEUTRALITY`。
##   如果皇道派路线判定未来变化，优先维护 `common/scripted_triggers/JAP_rework_scripted_triggers.txt`
##   中的 `Rance_is_jap_kodoha_ai`。
## - 写外交策略前先参考：
##   `common/ai_strategy/AI_Strategy_Reference/AI_diplomacy_strategy_reference.md`
##
## 日本自身策略模板：
##
## JAP_kodoha_example_strategy = {

## 	}
## 	enable = {
## 		Rance_is_jap_kodoha_ai = yes

## 基础亲德：皇道派日本应主动加入德国领导的轴心国。

	allowed = {
		original_tag = JAP
	}
	enable = {
		Rance_is_jap_kodoha_ai = yes
		country_exists = GER
		GER = {
			has_government = fascism
			NOT = { has_war_with = JAP }
		}
		NOT = { has_war_with = GER }
	}

	ai_strategy = {
		type = alliance
		id = GER
		value = 5000

	}

		type = send_volunteers_desire

		value = 500
	}
}

## 永久拒绝盟友战争牵连：皇道派日本不主动加入任何盟友战争，也不响应任何盟友的 call_allies。
JAP_kodoha_reject_all_ally_war_calls = {
	allowed = {
		always = yes
	}
	reversed = yes
	enable = {
		has_war = yes
		is_in_faction_with = JAP
		NOT = { tag = JAP }
	}
	enable_reverse = {
		JAP = {
			Rance_is_jap_kodoha_ai = yes
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = diplo_action_acceptance
		id = JAP
		target = call_allies
		value = -99999
	}
	ai_strategy = {
		type = diplo_action_desire
		id = JAP
		target = join_allies
		value = -99999
	}
}
