JAP_ai_land_production_update_equipment_flags = {
    JAP = {
        # Global production mode: expansion uses stockpile_ratio; replenishment uses deployed-army fill ratio.
        if = {
            limit = {
                has_country_flag = JAP_ai_land_production_replenishment_mode
                JAP_ai_land_production_should_stop_replenishment_mode = yes
            }
            clr_country_flag = JAP_ai_land_production_replenishment_mode
        }
        if = {
            limit = {
                NOT = { has_country_flag = JAP_ai_land_production_replenishment_mode }
                JAP_ai_land_production_should_start_replenishment_mode = yes
            }
            set_country_flag = JAP_ai_land_production_replenishment_mode
        }

        # infantry_equipment
        if = {
            limit = {
                has_country_flag = JAP_ai_land_production_infantry_equipment_increase
                JAP_ai_land_production_infantry_equipment_should_stop_increase = yes
            }
            clr_country_flag = JAP_ai_land_production_infantry_equipment_increase
        }
        if = {
            limit = {
                has_country_flag = JAP_ai_land_production_infantry_equipment_decline
                JAP_ai_land_production_infantry_equipment_should_stop_decline = yes
            }
            clr_country_flag = JAP_ai_land_production_infantry_equipment_decline
        }
        if = {
            limit = {
                NOT = { has_country_flag = JAP_ai_land_production_infantry_equipment_increase }
                NOT = { has_country_flag = JAP_ai_land_production_infantry_equipment_decline }
                JAP_ai_land_production_infantry_equipment_should_start_increase = yes
            }
            set_country_flag = JAP_ai_land_production_infantry_equipment_increase
        }
