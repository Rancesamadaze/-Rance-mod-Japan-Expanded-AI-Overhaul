JAP_apply_rance_ai_difficulty_easy_settings = {
	set_variable = { variable_rance_nandu = 1 }
	set_variable = { JAP_scripted_landing_raid_threshold = 15 }
	JAP_scripted_landing_refresh_raid_remaining = yes
	set_variable = { JAP_open_factory_max = 10 }
	if = {
		limit = { Rance_jap_return_to_peak_standard = yes }
		set_variable = { JAP_open_factory_max = 20 }
	}
}

JAP_apply_preselected_ai_difficulty_from_game_rule = {
	if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = EASY
			}
		}
		JAP_apply_rance_ai_difficulty_easy_settings = yes
	}
	else_if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = NORMAL
			}
		}
		JAP_apply_rance_ai_difficulty_normal_settings = yes
	}
	else_if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = HARD
			}
		}
		JAP_apply_rance_ai_difficulty_hard_settings = yes
	}
	else_if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = LUNATIC
			}
		}
		JAP_apply_rance_ai_difficulty_lunatic_settings = yes
	}
	else_if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = EXTRA
			}
		}
		JAP_apply_rance_ai_difficulty_extra_settings = yes
	}
	else_if = {
		limit = {
			has_game_rule = {
				rule = rance_japan_ai_difficulty
				option = PHANTASM
			}
		}
		JAP_apply_rance_ai_difficulty_phantasm_settings = yes
	}
}
