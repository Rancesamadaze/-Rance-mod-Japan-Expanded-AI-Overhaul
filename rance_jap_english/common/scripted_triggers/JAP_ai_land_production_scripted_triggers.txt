# JAP AI land production scripted triggers

# Expected scope: JAP country scope
# Current production model has two modes:
# - Expansion mode uses stockpile_ratio, i.e. stockpile / used equipment, where used includes deployed and training equipment.
# - Replenishment mode uses deployed-army fill ratio: (stockpile / 1000 + equipment in armies k) / (target equipment in armies k + 0.01).

# Replenishment mode gating: only high_demand equipment that is valid for current difficulty/design/stage participates.

JAP_ai_land_production_infantry_equipment_is_effective_high_demand = {
	always = yes
}

JAP_ai_land_production_mechanized_equipment_is_effective_high_demand = {
	Rance_nd_more_than_easy = yes
	JAP_has_cr_mechanized_equipment_design = yes

# Equipment production flag triggers. Names are consumed by JAP_ai_land_production_update_equipment_flags.
# Start triggers keep equipment availability gates; stop triggers remain ungated so existing flags can clear cleanly.

# infantry_equipment
JAP_ai_land_production_infantry_equipment_should_start_increase = {
	OR = {
		AND = {
			has_country_flag = JAP_ai_land_production_replenishment_mode
			set_temp_variable = { r_ratio = num_equipment@infantry_equipment }
			divide_temp_variable = { r_ratio = 1000 }
			add_to_temp_variable = { r_ratio = num_equipment_in_armies_k@infantry_equipment }
			set_temp_variable = { r_denom = num_target_equipment_in_armies_k@infantry_equipment }
			add_to_temp_variable = { r_denom = 0.01 }
			divide_temp_variable = { r_ratio = r_denom }

			check_variable = { r_ratio < JAP_ai_land_production_high_demand_start_increase }
		}
		AND = {
			NOT = { has_country_flag = JAP_ai_land_production_replenishment_mode }
			OR = {
				AND = {
					has_war = no
					stockpile_ratio = {
							archetype = infantry_equipment
							ratio < 0.12
						}
				}
				AND = {
					has_war = yes
					stockpile_ratio = {
							archetype = infantry_equipment
							ratio < 0.08
						}
				}
			}
		}
	}
}
