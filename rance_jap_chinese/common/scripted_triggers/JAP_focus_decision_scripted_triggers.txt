# Reusable gates for Japanese focus-decision workflows.

JAP_imperial_headquarters_us_mainland_invasion_route_ready = {
	OR = {
		AND = {
			Rance_is_jap_kodoha_ai = yes
			JAP_kodoha_major_asian_theaters_basic_missions_completed = yes
		}
		AND = {
			Rance_is_jap_lecture_ai = yes
			has_completed_focus = JAP_the_pacific_union_of_soviet_socialist_republics
			JAP_lecture_north_america_axis_capitals_controlled = yes
		}
		AND = {
			Rance_is_jap_kodoha_ai = no
			Rance_is_jap_lecture_ai = no
		}
	}
}
