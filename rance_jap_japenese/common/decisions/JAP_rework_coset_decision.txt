	JAP_rcs_player_cheat_buff_by_jap_ai_difficulty = {
		icon = GFX_decision_generic_decision
		visible = {
			NOT = {
				tag = JAP
			}
			JAP = {
				is_ai = yes
			}
			has_country_flag = JAP_setting_on
			date > 1936.2.1
		}
		available = {
			date > 1936.3.1
		}
		fire_only_once = yes
		complete_effect = {
			if = {
				limit = {
					NOT = { Rance_nd_more_than_easy = yes }
				}
				add_ideas = Rance_jap_ai_buff_easy
			}
			else_if = {
				limit = {
					Rance_nd_more_than_easy = yes
					NOT = { Rance_nd_more_than_normal = yes }
				}
				add_ideas = Rance_jap_ai_buff_normal
			}
			else_if = {
				limit = {
					Rance_nd_more_than_normal = yes
					NOT = { Rance_nd_more_than_hard = yes }
				}
				add_ideas = Rance_jap_ai_buff_hard
			}
			else_if = {
				limit = {
					Rance_nd_more_than_hard = yes
					NOT = { Rance_nd_more_than_lunatic = yes }
				}
				add_ideas = Rance_jap_ai_buff_lunatic
			}
			else_if = {
				limit = {
					Rance_nd_more_than_lunatic = yes
					NOT = { Rance_nd_more_than_extra = yes }
				}
				add_ideas = Rance_jap_ai_buff_extra
			}
			else = {
				add_ideas = Rance_jap_ai_buff_phantasm
			}
		}
