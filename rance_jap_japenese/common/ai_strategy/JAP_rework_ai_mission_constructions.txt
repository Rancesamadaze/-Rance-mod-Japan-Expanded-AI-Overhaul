# 在“首都圈开发计划”进行期间，用更高权重大致引导 AI 优先在关东补基建
JAP_shoko_develop_kanto_ai_infrastructure = {
	allowed = {
		original_tag = JAP
	}
	enable = {
		has_active_mission = JAP_shoko_develop_kanto_mission
		282 = {
			NOT = {
				OR = {
					infrastructure > 4
					check_variable = { building_level@infrastructure > JAP_shoko_kanto_infra_pre }
				}
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = build_building
		id = infrastructure
		target = 282
		value = 5000
	}
}
