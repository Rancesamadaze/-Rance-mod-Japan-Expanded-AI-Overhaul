# 实验：空天母舰增产策略
# 说明：当空天母舰净库存（总数 - 部署数）低于 100 时触发，开启高权重生产模式。
JAP_rework_mothership_production_increase = {
    allowed = {
        original_tag = JAP
    }

    enable = {
        # 必须完成特定特殊项目后启用 
        is_special_project_completed = sp:sp_air_mothership_aircraft
        num_of_military_factories > 200 # 工业门槛：确保只有当工业实力足够时才考虑增产
        # 触发阈值：净库存少于 100 架时启动紧急生产
        JAP_ai_air_production_mothership_below_low_threshold = yes
    }

    abort = {
        # 中止阈值：当净库存回升至 500 架以上时，停止此增产策略（与减产逻辑下限对接）
        JAP_ai_air_production_mothership_above_increase_stop_threshold = yes
    }

    ai_strategy = {
        type = equipment_variant_production_factor
        id = mothership_equipment
        value = 20
    }
    ai_strategy = {
		type = equipment_production_min_factories_archetype
		id = mothership_equipment
		value = 10
	}
}

# CAS 增产逻辑 (协同版)
# 触发：军工 > 100 且 战斗机充足 (> Rance_fighter_high_threshold) 且 CAS 极少 (< 1000)
JAP_rework_cas_production_increase = {
    allowed = { original_tag = JAP }

    enable = {
        # 工业门槛判定
        num_of_military_factories > 100
        
        # 战斗机总量判定 (需脱离增产阶段)
        JAP_ai_air_production_fighter_above_high_threshold = yes

        # CAS 总量判定
        JAP_ai_air_production_cas_below_low_threshold = yes
    }

    abort = {
        # 中止判定：CAS 补齐(>2500) OR 战斗机告急(< Rance_fighter_low_threshold) OR 工业滑坡
        or = {
            JAP_ai_air_production_cas_above_increase_stop_threshold = yes
            JAP_ai_air_production_fighter_below_low_threshold = yes
            num_of_military_factories < 101
        }
    }

    ai_strategy = {
        type = equipment_production_factor
        id = cas
        value = 30 # 根据要求调整为 30
    }
    ai_strategy = {
        type = equipment_production_min_factories
        id = cas
        value = 30 # 根据要求调整为 30
    }
}

# CAS 减产逻辑 (协同版)
# 触发：CAS 规模过大(>6000) OR 战斗机急需增产(< Rance_fighter_low_threshold)
JAP_rework_cas_production_reduce = {
    allowed = { original_tag = JAP }

    enable = {
        or = {
            JAP_ai_air_production_cas_above_high_threshold = yes
            JAP_ai_air_production_fighter_below_low_threshold = yes
        }
    }

    abort = {
        # 中止判定：CAS 跌破 4000 且 战斗机已恢复 (> Rance_fighter_high_threshold)
        and = {
            JAP_ai_air_production_cas_below_reduce_stop_threshold = yes
            JAP_ai_air_production_fighter_above_high_threshold = yes
        }
    }

    ai_strategy = {
        type = equipment_production_factor
        id = cas
        value = -99
    }
    ai_strategy = {
        type = equipment_production_min_factories
        id = cas
        value = -99 # 强制撤走保底工厂，确保资源回流轻战
    }
    ai_strategy = {
        type = build_airplane
        id = cas
        value = -99999
    }
}
