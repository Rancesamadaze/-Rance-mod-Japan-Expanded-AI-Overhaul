##easy
country_event = {
    id = Rance_jap_nandu_event.1
    picture = GFX_rance_nandu_set
    title = Rance_jap_nandu_event.1.t
    desc = Rance_jap_nandu_event.1.d
    is_triggered_only = yes
    option = {
        name = Rance_jap_nandu_event.1.d.a 
        hidden_effect = {
            JAP = {
                set_variable = {
                    variable_rance_nandu = 1
                }
                set_variable = {
                    JAP_scripted_landing_raid_threshold = 15
                }
                JAP_scripted_landing_refresh_raid_remaining = yes
                set_variable = {
                    JAP_open_factory_max = 10
                }
                if = {
                    limit = { Rance_jap_return_to_peak_standard = yes }
                    set_variable = {
                        JAP_open_factory_max = 20
                    }
                }
            }
        }
    }
    option = {
        name = Rance_jap_nandu_event.r
        country_event = Rance_jap_event.1
    }
}
